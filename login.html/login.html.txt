  <script>
    function toggleForm() {
      const container = document.getElementById('container');
      const loginForm = document.getElementById('loginForm');
      const signupForm = document.getElementById('signupForm');
      
      container.classList.toggle('sign-up-mode');
      
      if (container.classList.contains('sign-up-mode')) {
        loginForm.style.display = 'none';
        signupForm.style.display = 'flex';
        document.getElementById('bannerTitle').textContent = "Let's Start Learning!";
        document.getElementById('bannerText').textContent = "Create your account and begin your HTML journey today.";
      } else {
        loginForm.style.display = 'flex';
        signupForm.style.display = 'none';
        document.getElementById('bannerTitle').textContent = "Hello, Learner!";
        document.getElementById('bannerText').textContent = "Join thousands of students mastering HTML with interactive lessons.";
      }
    }

    function handleLogin() {
      alert("🔐 Login functionality will connect with Firebase soon!");
      // পরে Firebase auth যোগ করব
    }

    function handleSignup() {
      alert("🎉 Account creation will connect with Firebase soon!");
    }

    // Enter key support
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Enter') {
        const activeForm = document.querySelector('.form-container[style*="display: flex"]');
        if (activeForm) {
          if (activeForm.id === 'loginForm') handleLogin();
          else handleSignup();
        }
      }
    });
  </script>
